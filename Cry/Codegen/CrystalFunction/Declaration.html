def source
  raise 'name is not set' unless @name
  raise 'size of arg_names and arg_types are not same' unless @arg_names.size == @arg_types.size

  args = arg_names.zip(arg_types).map do |n, t|
    "#{n} : #{t}"
  end.join(', ')
  "#{@decl} #{@name}(#{args}) : #{@ret_type}"
end
